292  DeviceState deviceState(uint16_t status_word)
293  {
294  if ((status_word & 0b01001111) == 0b00000000){
295  return STATE_NOT_READY_TO_SWITCH_ON;
296  }
297  else if ((status_word & 0b01001111) == 0b01000000){
298  return STATE_SWITCH_ON_DISABLED;
299  }
300  else if ((status_word & 0b01101111) == 0b00100001){
301  return STATE_READY_TO_SWITCH_ON;
302  }
303  else if ((status_word & 0b01101111) == 0b00100011){
304  return STATE_SWITCH_ON;
305  }
306  else if ((status_word & 0b01101111) == 0b00100111){
307  if(mode_of_operation_display_ == MODE_HOMING){
308  if ((((status_word & 0xff00)>>8) & 0b00110100 ) == 0b00000000)
309  return STATE_HOMING_PROGRESS;
310  else if((((status_word & 0xff00)>>8) & 0b00110100 ) == 0b00000100)
311  return STATE_HOMING_NOT_START;
312  else if((((status_word & 0xff00)>>8) & 0b00110100 ) == 0b00010000)
313  return STATE_HOMING_ATTAINED_NOT_REACHED;
314  else if((((status_word & 0xff00)>>8) & 0b00110100 ) == 0b00010100){
315  homing_ = true;
316  return STATE_HOMING_COMPLITE;
317  }
318  else if((((status_word & 0xff00)>>8) & 0b00110100 ) == 0b00100000)
319  return STATE_HOMING_ERROR;
320  else if((((status_word & 0xff00)>>8) & 0b00110100 ) == 0b00100100)
321  return STATE_HOMING_ERROR;
322  else
323  return STATE_HOMING_UNDIFINED;
324  }
325  else{
326  return STATE_OPERATION_ENABLED;
327  }
328  }
329  else if ((status_word & 0b01101111) == 0b00000111){
330  return STATE_QUICK_STOP_ACTIVE;
331  }
332  else if ((status_word & 0b01001111) == 0b00001111){
333  return STATE_FAULT_REACTION_ACTIVE;
334  }
335  else if ((status_word & 0b01001111) == 0b00001000){
336  return STATE_FAULT;
337  }
338  return STATE_UNDEFINED;
339  }

341  uint16_t transition(DeviceState state, uint16_t control_word)
342  {
343  switch(state)
344  {
345  case STATE_START: // -> STATE_NOT_READY_TO_SWITCH_ON (automatic)
346  return control_word;
347  case STATE_NOT_READY_TO_SWITCH_ON: // -> STATE_SWITCH_ON_DISABLED (automatic)
348  return control_word;
349  case STATE_SWITCH_ON_DISABLED: // -> STATE_READY_TO_SWITCH_ON
350  return ((control_word & 0b01111110) | 0b00000110);
351  case STATE_READY_TO_SWITCH_ON: // -> STATE_SWITCH_ON
352  return ((control_word & 0b01110111) | 0b00000111);
353  case STATE_SWITCH_ON: // -> STATE_OPERATION_ENABLED
354  return ((control_word & 0b01110000) | 0b00001111);
355  case STATE_OPERATION_ENABLED: // -> GOOD
356  return ((control_word & 0b00000000) | 0b00001111);
357  case STATE_QUICK_STOP_ACTIVE: // -> STATE_OPERATION_ENABLED
358  return ((control_word & 0b01111111) | 0b00001111);
359  case STATE_FAULT_REACTION_ACTIVE: // -> STATE_FAULT (automatic)
360  return control_word;
361  case STATE_FAULT: // -> STATE_SWITCH_ON_DISABLED
362  return ((control_word & 0b11111111) | 0b10000000);
363  case STATE_HOMING_PROGRESS:
364  return ((control_word & 0b00011111) | 0b00011111);
365  case STATE_HOMING_NOT_START:
366  if(homing_ready_)
367  return ((control_word & 0b00011111) | 0b00011111);
368  else
369  return control_word;
370  case STATE_HOMING_ATTAINED_NOT_REACHED:
371  return ((control_word & 0b00011111) | 0b00011111);
372  case STATE_HOMING_COMPLITE:
373  return ((control_word & 0b00001111) | 0b00001111);
374  case STATE_HOMING_ERROR:
375  return ((control_word & 0b00001111) | 0b00001111);
376  case STATE_HOMING_UNDIFINED:
377  return ((control_word & 0b00011111) | 0b00011111);
378  default:
379  break;
380  }
381  return control_word;
382  }
